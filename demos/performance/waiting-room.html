function loadWaitingRoomDemoDemo() {
    console.log("Loading Waiting Room Demo demo...");
    // Initialize demo components
    const demoContainer = document.getElementById("demo-container");
    demoContainer.innerHTML = "Demo is loading...";
    
    // Connect to demo API
    fetch("https://automatic-demo.com/waiting-room-demo")
        .then(response => {
            console.log("Demo loaded successfully");
            demoContainer.innerHTML = "Demo is ready!";
        })
        .catch(error => {
            console.error("Error loading demo:", error);
        });
}

// Call the function when page loads
document.addEventListener("DOMContentLoaded", loadWaitingRoomDemoDemo);
